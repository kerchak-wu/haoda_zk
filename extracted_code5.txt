from mpython import *
import framebuf
import font.digiface_30
import time
from machine import Timer

def display_font(_font, _str, _x, _y, _wrap, _z=0):
    _start = _x
    for _c in _str:
        _d = _font.get_ch(_c)
        if _wrap and _x > 128 - _d[2]: _x = _start; _y += _d[1]
        if _c == '1' and _z > 0: oled.fill_rect(_x, _y, _d[2], _d[1], 0)
        oled.blit(framebuf.FrameBuffer(bytearray(_d[0]), _d[2], _d[1],
        framebuf.MONO_HLSB), (_x+int(_d[2]/_z)) if _c=='1' and _z>0 else _x, _y)
        _x += _d[2]

def on_button_a_down(_):
    global step, flag
    time.sleep_ms(10)
    if button_a.value() == 1: return
    flag = 1

def on_button_b_down(_):
    global step, flag
    time.sleep_ms(10)
    if button_b.value() == 1: return
    flag = 0

_is_shaked = _is_thrown = False
_last_x = _last_y = _last_z = _count_shaked = _count_thrown = 0
def on_shaked():pass
def on_thrown():pass

tim11 = Timer(11)
def timer11_tick(_):
    global _is_shaked, _is_thrown, _last_x, _last_y, _last_z, _count_shaked, _count_thrown
    if _is_shaked:
        _count_shaked += 1
        if _count_shaked == 5: _count_shaked = 0
    if _is_thrown:
        _count_thrown += 1
        if _count_thrown == 10: _count_thrown = 0
        if _count_thrown > 0: return
    x=accelerometer.get_x(); y=accelerometer.get_y(); z=accelerometer.get_z()
    _is_thrown = (x * x + y * y + z * z < 0.25)
    if _is_thrown: on_thrown();return
    if _last_x == 0 and _last_y == 0 and _last_z == 0:
        _last_x = x; _last_y = y; _last_z = z; return
    diff_x = x - _last_x; diff_y = y - _last_y; diff_z = z - _last_z
    _last_x = x; _last_y = y; _last_z = z
    if _count_shaked > 0: return
    _is_shaked = (diff_x * diff_x + diff_y * diff_y + diff_z * diff_z > 1)
    if _is_shaked: on_shaked()

tim11.init(period=100, mode=Timer.PERIODIC, callback=timer11_tick)

def on_shaked():
    global step, flag
    if flag == 1:
        step = step + 1

button_a.irq(trigger=Pin.IRQ_FALLING, handler=on_button_a_down)
button_b.irq(trigger=Pin.IRQ_FALLING, handler=on_button_b_down)

flag = 0
step = 0
while True:
    if flag == 0:
        oled.fill(0)
        oled.DispChar(str('计步器'), 0, 0, 1)

        oled.DispChar(str('按下B键步数清零'), 0, 48, 1)
        oled.show()
        step = 0
    elif flag == 1:
        oled.fill(0)
        oled.DispChar(str('步数'), 0, 15, 1)
        display_font(font.digiface_30, str((str(step))), 30, 10, False, 2)

            oled.DispChar(str((''.join([str(x) for x in ['还差', 7000 - step, '，加油加油']]))), 0, 48, 1)
            oled.show()

from mpython import *
import time

def Gradually_brighten():
    global blue_breath, green_breath, red_breath
    while not red_breath > 255:
        while not green_breath > 255:
            while not blue_breath > 255:
                rgb.fill( (int(red_breath), int(green_breath), int(blue_breath)) )
                rgb.write()
                time.sleep_ms(1)
                blue_breath = blue_breath + 1
            rgb.fill( (int(red_breath), int(green_breath), int(blue_breath)) )
            rgb.write()
            time.sleep_ms(1)
            green_breath = green_breath + 1
        rgb.fill( (int(red_breath), int(green_breath), int(blue_breath)) )
        rgb.write()
        time.sleep_ms(1)
        red_breath = red_breath + 1

def Gradually_darken():
    global blue_breath, green_breath, red_breath
    while not red_breath < 0:
        while not green_breath < 0:
            while not blue_breath < 0:
                rgb.fill( (int(red_breath), int(green_breath), int(blue_breath)) )
                rgb.write()
                time.sleep_ms(1)
                blue_breath = blue_breath + (-1)
            rgb.fill( (int(red_breath), int(green_breath), int(blue_breath)) )
            rgb.write()
            time.sleep_ms(1)
            green_breath = green_breath + (-1)
        rgb.fill( (int(red_breath), int(green_breath), int(blue_breath)) )
        rgb.write()
        time.sleep_ms(1)
        red_breath = red_breath + (-1)

red_breath = 0
green_breath = 0
blue_breath = 0
while True:
    Gradually_brighten()
    Gradually_darken()

from mpython import *
import framebuf
import font.digiface_44

def display_font(_font, _str, _x, _y, _wrap, _z=0):
    _start = _x
    for _c in _str:
        _d = _font.get_ch(_c)
        if _wrap and _x > 128 - _d[2]: _x = _start; _y += _d[1]
        if _c == '1' and _z > 0: oled.fill_rect(_x, _y, _d[2], _d[1], 0)
        oled.blit(framebuf.FrameBuffer(bytearray(_d[0]), _d[2], _d[1],
        framebuf.MONO_HLSB), (_x+int(_d[2]/_z)) if _c=='1' and _z>0 else _x, _y)
        _x += _d[2]
p0 = MPythonPin(0, PinMode.ANALOG)
p1 = MPythonPin(1, PinMode.OUT)

while True:
    oled.fill(0)
    display_font(font.digiface_44, str(p0.read_analog()), 0, 0, False, 2)
    oled.show()
    # 光线较暗时
    if p0.read_analog() < 1000:
        # 打开灯光
        p1.write_digital(1)
    else:
        # 关闭灯光
        p1.write_digital(0)
